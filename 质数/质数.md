```cpp
//朴素法
#include<iostream>
#include<cstdio>
#include<cmath>
using namespace std;
int n;
bool isPrime(int i){
    for(int j =2;j<=sqrt(i);j++){
        if(i%j==0)return false;
    }
    return true;
}
int main(){
    cin>>n;
    for(int i =2;i<=n;i++){
        if(isPrime(i)){
            cout<<i<<" ";
        }    
    }

}
```
